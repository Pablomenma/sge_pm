    maximo = lista[0]
    minimo = lista[0]

    for num in lista:
        if num > maximo:
            maximo = num
        if num < minimo:

    return maximo, minimo

### Concatenar listas: Dadas dos listas de números, crea una función que devuelva una tercera lista que contenga los elementos de ambas listas intercalados.
```python
numeros_a  = [1, 2, 3, 4, 5]
numeros_b = [12, 23, 34, 45, 56]

def concatenador(numeros_a, numeros_b):
    salida = []
    longitud_result = len(numeros_a) if len(numeros_a) < len(numeros_b) else len(numeros_b)
    for i in range(longitud_result):
        salida.append(numeros_a[i])
        salida.append(numeros_b[i])
    return salida

print(concatenador(numeros_a, numeros_b))
```
